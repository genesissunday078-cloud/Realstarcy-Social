<script>
  function show(id) { document.querySelectorAll('.view').forEach(v => v.classList.remove('active')); document.getElementById(id).classList.add('active'); }
  
  // Tabs & Swipe
  const feedsList = ['realstarcy-content', 'following-content', 'foryou-content'];
  const tabBtns = ['btn-realstarcy', 'btn-following', 'btn-foryou'];
  let currentFeedIndex = 2;
  function showFeed(id) { 
    currentFeedIndex = feedsList.indexOf(id);
    document.querySelectorAll('.feed').forEach(f => f.style.display = 'none'); 
    document.getElementById(id).style.display = 'block'; 
  }

  // Like System
  function likeVideo(btn) {
    btn.classList.toggle('liked');
    let countSpan = btn.querySelector('.like-count');
    let val = parseInt(countSpan.innerText.replace(/,/g, ''));
    countSpan.innerText = (btn.classList.contains('liked') ? val + 1 : val - 1).toLocaleString();
    btn.innerHTML = (btn.classList.contains('liked') ? '❤️ ' : '🤍 ') + '<span class="like-count">' + countSpan.innerText + '</span>';
  }
  function doubleClickLike(card) {
    let likeBtn = card.querySelector('.like-btn');
    if (likeBtn && !likeBtn.classList.contains('liked')) likeVideo(likeBtn);
    let heart = document.createElement('div');
    heart.className = 'heart-animation animate-heart'; heart.innerHTML = '❤️';
    card.appendChild(heart); setTimeout(() => heart.remove(), 800);
  }

  // Profile Logic
  function uploadAvatar(e) { const reader = new FileReader(); reader.onload = function() { document.getElementById('avatar').src = reader.result; }; reader.readAsDataURL(e.target.files[0]); }
  function saveProfile() {
    document.getElementById('viewName').innerText = document.getElementById('inName').value || 'Genesis';
    document.getElementById('viewHandle').innerText = '@' + (document.getElementById('inHandle').value || 'Genedino');
    document.getElementById('viewBio').innerText = document.getElementById('inBio').value || 'Musician, Artisan & Engineer';
    document.getElementById('editModal').style.display = 'none';
  }

  // Camera
  async function openCam() { document.getElementById('cam-view').style.display = 'block'; try { const s = await navigator.mediaDevices.getUserMedia({video: true, audio: true}); document.getElementById('v').srcObject = s; } catch(e) { alert("Camera access needed"); } }
  function closeCam() { document.getElementById('cam-view').style.display = 'none'; const s = document.getElementById('v').srcObject; if(s) s.getTracks().forEach(t => t.stop()); }
</script>
